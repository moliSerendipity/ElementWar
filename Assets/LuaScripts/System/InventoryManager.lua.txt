-- 负责处理物品的增删改查
local Config_Item = require("Config.Config_Item")
local ItemData = require("Data.ItemData")

local InventoryManager = {}

-- 静态变量
InventoryManager.MaxSlots = 30				-- 背包总格子数
InventoryManager.Slots = {}					-- 背包数据 (Model): key = 索引(1到30), value = ItemData实例
local weaponUIDCounter = 10000    			-- 模拟服务器发放的全局唯一 ID 生成器

-- 初始化背包
function InventoryManager.Init()
	for i = 1, InventoryManager.MaxSlots do
		InventoryManager.Slots[i] = nil
	end
	print("[InventoryManager] Initialized with", InventoryManager.MaxSlots, "slots.")
end

-- 添加物品 (考虑堆叠)
-- 返回值: 成功添加的数量 (可能因为背包满了只装下一部分)
function InventoryManager.AddItem(_configID, _count)
	local config = Config_Item[_configID]
	if not config then
		return 0
	end

	local leftCount = _count

	-- 如果是可堆叠物品，先找有没有没装满的同类格子
	if config.MaxStack > 1 then
		for i = 1, InventoryManager.MaxSlots do
			local item = InventoryManager.Slots[i]
			if item and item.ConfigID == _configID and item.Count < config.MaxStack then
				local leftSpace = config.MaxStack - item.Count 		-- 当前格子还能装多少
				if leftCount <= leftSpace then
					item.Count = item.Count + leftCount
					return _count 		-- 全部存入
				else
					item.Count = config.MaxStack			-- 填满这个格子
					leftCount = leftCount - leftSpace		-- 剩下的继续找下一个格子
				end
			end
		end
	end

	-- 如果还有剩余，或者是不重叠物品（武器），找空格子放
	while leftCount > 0 do
		local isEmptySlotIndex = InventoryManager.GetFirstEmptySlot()
		if isEmptySlotIndex == -1 then
			print(string.format("[InventoryManager] 警告：背包已满！%d 个 %s 被丢弃。", leftCount, config.Name))
			break
		end

		local addCount = math.min(leftCount, config.MaxStack)

		-- 生成 UID (武器给唯一ID，材料给0)
        local uid = 0
        if config.Type == 1 then
            weaponUIDCounter = weaponUIDCounter + 1
            uid = weaponUIDCounter
        end

		InventoryManager.Slots[isEmptySlotIndex] = ItemData.New(_configID, addCount, uid)
		leftCount = leftCount - addCount
	end

	-- TODO: 触发事件通知 UI 刷新
    -- EventManager.Fire("OnInventoryChanged")

    local actualAdded = _count - leftCount
    print(string.format("[InventoryManager] 成功添加 %d 个 %s", actualAdded, config.Name))
    return actualAdded
end

-- 保持物品状态添加至背包
function InventoryManager.AddExistingItem(_itemData)
    local emptyIndex = InventoryManager.GetFirstEmptySlot()
    if emptyIndex ~= -1 then
        InventoryManager.Slots[emptyIndex] = _itemData
        return true
    end
    print("[InventoryManager] 背包已满，无法退回原有装备！")
    return false
end

-- 寻找第一个空格子
function InventoryManager.GetFirstEmptySlot()
	for i = 1, InventoryManager.MaxSlots do
		if InventoryManager.Slots[i] == nil then
			return i
		end
	end

	return -1		-- 背包已满
end

-- 扣除物品
-- _isForced: 当背包里需要扣除的物品数量不足时，是否强制扣除该物品直到数量为0
function InventoryManager.ConsumeItem(_configID, _count, _isForced)
	local currentTotal = InventoryManager.GetItemCount(_configID)
	if (not _isForced and currentTotal < _count) or currentTotal == 0 then
		print(string.format("[InventoryManager] 扣除失败：[%d] 数量不足 (拥有:%d, 需要:%d)", _configID, currentTotal, _count))
        return false, 0
    end

    local needToConsume = _count
    local actualConsumed = 0
    for i = 1, InventoryManager.MaxSlots do
    	local item = InventoryManager.Slots[i]
    	if item and item.ConfigID == _configID then
    		if item.Count >= needToConsume then
    			item.Count = item.Count - needToConsume
    			actualConsumed = actualConsumed + needToConsume
    			if item.Count == 0 then
    				InventoryManager.Slots[i] = nil
    			end
    			-- TODO: 触发 UI 刷新
    			return true, actualConsumed
    		else
    			needToConsume = needToConsume - item.Count
    			actualConsumed = actualConsumed + item.Count
    			InventoryManager.Slots[i] = nil
    		end
    	end
    end

    return true, actualConsumed
end

-- 获取某物品总数
function InventoryManager.GetItemCount(_configID)
	local total = 0
	for i = 1, InventoryManager.MaxSlots do
		local item = InventoryManager.Slots[i]
		if item and item.ConfigID == _configID then
			total = total + item.Count
		end
	end

	return total
end

-- Debug: 打印背包内容
function InventoryManager.PrintInventory()
	print("\n=============== 🎒 当前背包状态 ===============")
	local isEmpty = true

	for i = 1, InventoryManager.MaxSlots do
		if InventoryManager.Slots[i] ~= nil then
			print(string.format("格子[%02d]: %s", i, InventoryManager.Slots[i]:ToString()))
			isEmpty = false
		end
	end

	if isEmpty then
		print("背包是空的")
	end
	print("===============================================\n")
end


return InventoryManager
