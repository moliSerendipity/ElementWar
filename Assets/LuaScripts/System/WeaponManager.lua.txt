-- 负责处理战斗中的武器状态与子弹切换逻辑

local Config_Common = require("Config.Config_Common")
local InventoryManager = require("System.InventoryManager")
local CombatManager = require("System.CombatManager")
local TeamManager = require("System.TeamManager") 

local WeaponManager = {}

function WeaponManager.Update(_deltaTime)
	local activeCharacter = TeamManager.GetCurrentPlayer()
	if not activeCharacter then
		return
	end

	local weapon = activeCharacter:GetActiveWeapon()
	if not weapon then
		return
	end

	local weaponSubType = weapon.Config.WeaponSubType
	-- 只处理枪械和近战，投掷物走单独的按键或立刻释放，不在 Update 里轮询
	if weaponSubType ~= Config_Common.WeaponSubType.Ranged and weaponSubType ~= Config_Common.WeaponSubType.Melee then
		return
	end

	-- 更新冷却
	if weapon.FireCooldown > 0 then
		weapon.FireCooldown = weapon.FireCooldown - _deltaTime
	end

	local inputFrame = CS.InputManager.Instance.Frame
	local attemptToFire = false

	-- 判断意图
	if weaponSubType == Config_Common.WeaponSubType.Ranged then
		if weapon.Config.IsAuto then
			attemptToFire = inputFrame.fireButton.isHeld
		else
			attemptToFire = inputFrame.fireButton.wasPressedThisFrame
		end
	elseif weaponSubType == Config_Common.WeaponSubType.Melee then
		attemptToFire = inputFrame.fireButton.isHeld
	end

	-- 意图与冷却碰撞
	if attemptToFire and weapon.FireCooldown <= 0 then
		local success = CombatManager.Fire(activeCharacter)

		if success then
			-- 成功执行动作，重置冷却
            if weaponSubType == Config_Common.WeaponSubType.Ranged then
                weapon.FireCooldown = weapon.Config.FireRate
            elseif weaponSubType == Config_Common.WeaponSubType.Melee then
                weapon.FireCooldown = weapon.Config.AttackRate
            end
        else
            weapon.FireCooldown = 0.2
        end
    end
end

function WeaponManager.Init()
	-- XLua 绑定 C# 事件的语法：CS.命名空间.类名.Instance:事件名('+', 回调函数)
	if CS.InputManager.Instance then
		CS.InputManager.Instance:OnSwitchAmmoType('-', WeaponManager.OnSwitchAmmoTypePressed)
        CS.InputManager.Instance:OnSwitchAmmoType('+', WeaponManager.OnSwitchAmmoTypePressed)
		print("[WeaponManager] 成功监听 C# 的切换子弹按键事件")
    else
        print("[Error] InputManager C# 实例未找到！")
    end
end

function WeaponManager.OnSwitchAmmoTypePressed()
	-- 获取当前主控角色
	local activeCharacter = TeamManager.GetCurrentPlayer()
	if not activeCharacter then
		return
	end

	-- 获取当前手持武器
	local weapon = activeCharacter:GetActiveWeapon()
	if not weapon or not weapon.Config.WeaponSubType or weapon.Config.WeaponSubType ~= Config_Common.WeaponSubType.Ranged then
		print("[WeaponManager] 无法切弹")
		return
	end

	-- 获取下一种可用子弹
	local nextAmmoID = WeaponManager.GetNextAvailableAmmo(weapon)

	if nextAmmoID and nextAmmoID ~= weapon.CurrentAmmoType then
		local success = CombatManager.SwitchAmmoType(activeCharacter, nextAmmoID)
		-- 如果逻辑执行成功，反向调用 C# 播放动画
		if success then
			local tpsController = CS.TeamController.Instance:GetCurrentPlayer()
			if tpsController then
				tpsController:ForceReload()
			end
		end
	else
		print("[WeaponManager] 背包中没有其他适用当前枪械的元素子弹。")
    end
end

-- 在背包中寻找下一种可用的子弹
function WeaponManager.GetNextAvailableAmmo(_weapon)
	local currentAmmoID = _weapon.CurrentAmmoType
	local weaponClass = _weapon.Config.WeaponClass

	-- 根据配置表规则计算该枪械对应的基础子弹 ID
    -- 公式: AmmoIDStart(3000) + (WeaponClass * 100)
    local baseAmmoID = Config_Common.AmmoIDStart + (weaponClass * 100)

    -- 计算当前子弹的偏移量 (0=物理, 1=火, 2=雷, 3=冰)
    local currentOffset = currentAmmoID % 100

    -- 循环向后查找元素
    local maxElements = 4
    for i = 1, maxElements do
    	local nextOffset = (currentOffset + i) % maxElements
    	local checkAmmoID = baseAmmoID + nextOffset
    	-- 如果算出的这个 ID 在背包里有库存
    	if checkAmmoID ~= currentAmmoID and InventoryManager.GetItemCount(checkAmmoID) > 0 then
    		return checkAmmoID
    	end
    end

	-- 找了一圈都没有，返回原本的子弹
    return currentAmmoID
end

function WeaponManager.OnFirePressed()
	local activeCharacter = TeamManager.GetCurrentPlayer()
	if not activeCharacter then
		return
	end

	local success = CombatManager.Fire(activeCharacter)
end

-- 清理函数：注销所有 C# 事件
function WeaponManager.Dispose()
	if CS.InputManager.Instance then
		CS.InputManager.Instance:OnSwitchAmmoType('-', WeaponManager.OnSwitchAmmoTypePressed)
        print("[WeaponManager] 成功解绑 C# 的 OnSwitchAmmoType 事件")
    end
end


return WeaponManager
