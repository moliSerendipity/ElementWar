-- 合并所有配置
local Config_Common = require("Config.Config_Common")
local Config_Weapon = require("Config.Config_Weapon")
local Config_Ammo = require("Config.Config_Ammo")
local Config_Throwable = require("Config.Config_Throwable")
local Config_Consumable = require("Config.Config_Consumable")

local Config_Item = {}

-- 合并武器配置
for id, config in pairs(Config_Weapon) do
    Config_Item[id] = config
end

-- 合并弹药配置
for id, config in pairs(Config_Ammo) do
    Config_Item[id] = config
end

-- 合并投掷物配置
for id, config in pairs(Config_Throwable) do
    Config_Item[id] = config
end

-- 合并消耗品配置
for id, config in pairs(Config_Consumable) do
    Config_Item[id] = config
end

-- 设置配置表为只读（防止运行时篡改）
local readonlyMeta = {
    __newindex = function(t, k, v)
        error("尝试修改只读配置表: " .. tostring(k))
    end,
    __metatable = "只读配置表"
}

setmetatable(Config_Item, readonlyMeta)

return Config_Item
