-- 配置校验函数
local function ValidateWeaponConfig()
	for id, config in pairs(Config_Weapon) do
		-- 检查必填字段
		local requiredFields = { "ID", "Name", "Type", "WeaponClass", "EquipSlot", "MaxStack", "BaseDamage"}
		for _, field in ipairs(requiredFields) do
			if config[field] == nil then
				error(string.format("武器配置 %d 缺少必填字段: %s", id, field))
			end
		end
		-- TODO: 检查ID是否符合规则
	end
	print("[配置校验] 武器配置校验通过!")
end

ValidateWeaponConfig()

return Config_Weapon
