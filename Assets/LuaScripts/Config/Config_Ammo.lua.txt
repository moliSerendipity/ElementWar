-- 配置校验
local function ValidateAmmoConfig()
    for id, config in pairs(Config_Ammo) do
        local requiredFields = { "ID", "Name", "Type", "WeaponClass", "ElementType", "DamageMultiplier", "MaxStack" }
        for _, field in ipairs(requiredFields) do
            if config[field] == nil then
                error(string.format("弹药配置 %d 缺少必填字段: %s", id, field))
            end
        end
        -- TODO: 检查ID是否符合规则
    end
    print("[配置校验] 弹药配置校验通过!")
end
